C = 20.0 # SVM regularisation parameter
n = 1000 # number of iterations

# primal linear decision function
def df_pr(x,w):
    return np.squeeze(np.dot(w.T,x))

# hinge loss
def hinge_pr(w):
    loss = 0.0 # init
    x = np.zeros((3,1))
    for i in range(ntrain):
        x = np.reshape(X1[i,:], (3,1))
        loss += max(0.0, 1.0 - y0[i]*df_pr(x,w))
    return loss

# Pegasos
x = np.zeros((3,1))
loss_pr = np.zeros(n)
wlen_pr = np.zeros(n)
w_pr = np.zeros((3,1)) # initial weight vector
for i in range(n):
    for j in range(ntrain):
        eta = 1.0/(1.0 + i)
        x = np.reshape(X1[j,:], (3,1))
        yf = y0[j]*df_pr(x,w_pr)
        if yf < 1:
            w_pr = (1 - eta)*w_pr + eta*C*y0[j]*x
        else:
            w_pr = (1 - eta)*w_pr
    loss_pr[i] = hinge_pr(w_pr)
    wlen_pr[i] = LA.norm(w_pr)
        
print('Final loss:', loss_pr[-1], '  Final length of w:', wlen_pr[-1])
plt.plot(loss_pr);

def df_dual(x,a,eta):

    return np.squeeze(np.dot(C*eta*(a*y0).T,ccv))

    loss = 0.0 # init
    for i in range(ntrain):
        x = np.reshape(X1[i,:], (3,1))

2. Nichtlineare Supportvektormaschine¶

Die Eingangsdaten kommen nur als Skalarprodukte vor $\Rightarrow$ Kernel-Trick anwendbar: ersetze alle Skalarprodukte $ \langle {\bf x}_i \cdot {\bf x}_j \rangle$ durch eine gültige Kernfunktion $ k( {\bf x}_i, {\bf x}_j)$:

Dafür benötigen wir eine Kernfunktion. Im folgenden definieren wir einen polynomiellen und einen RBF-Kern:

  • Homogener Polynomkern: $$k( {\bf x}, {\bf y}) = ({\bf x}^\top {\bf y})^p $$

  • RBF-Kern: $$k( {\bf x}, {\bf y}) = e^{-\frac{\|{\bf x} - {\bf y}\|^2}{2\sigma^2}}$$

In [27]:
# Homogener Polynomkern
def hpoly(x, y):
    
    # Grammatrix
    G = np.dot(x, y.T)
    
    # Kernelmatrix
    deg = 5
    K = G**deg
    
    return K

    
# RBF - Kern
def rbf(x, y):
    
    # quadratische Distanzmatrix
    if x.ndim == 1:
        D = np.sum((y - x)**2, axis=1)
    else:
        D = distance.cdist(x, y, 'euclidean')
        D = D**2
    
    # Kernelmatrix
    rho = 0.3
    K = np.exp(-0.5/(rho**2)*D)
    
    return K

# Allg. Kernfunktion
def k(x,y):
    
    #return hpoly(x,y)
    return rbf(x, y)

Für das Minimierungsproblem müssen wir nur die Grammatrix durch die Kernelmatrix $K_{ij} = k( {\bf x}_i, {\bf x}_j) $ ersetzen, sonst bleibt alles gleich:

$$ P = (\mathbf{y} \cdot \mathbf{y}^\top) \otimes K $$

Wie im linearen Fall haben die Supportvektoren $\alpha_i > 0$, allerdings können wir jetzt den unendlichdimensionalen Gewichtsvektor ${\bf w}^*$ nicht mehr berechnen. Trennbreite der Hyperebene (Achtung: die Trennbreite ist kein Abstand im Eingaberaum, sondern im hochdimensionalen Merkmalsraum):

$$\gamma = \frac{1}{\| {\bf w}^* \|} \quad \text{mit} \quad \langle {\bf w} \cdot {\bf w} \rangle = \sum_{i,j=1}^\ell y_i y_j \alpha_i \alpha_j \langle \Phi({\bf x}_i) \cdot \Phi({\bf x}_j) \rangle = \sum_{i,j=1}^\ell y_i y_j \alpha_i \alpha_j k( {\bf x}_i, {\bf x}_j) $$
